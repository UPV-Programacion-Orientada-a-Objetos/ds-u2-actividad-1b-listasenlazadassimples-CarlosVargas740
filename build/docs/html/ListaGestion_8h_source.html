29/**
30 * @class ListaGestion
31 * @brief Lista enlazada de sensores con gestión polimórfica

33 * Almacena punteros a SensorBase, permitiendo procesamiento

36class ListaGestion {
37private:

41 /**
42 * @brief Libera memoria de todos los sensores
43 */
44 void liberar() {
45 NodoSensor* actual = cabeza;
46 while (actual != nullptr) {
47 NodoSensor* temp = actual;
48 actual = actual->siguiente;
49 delete temp->sensor; // Llama al destructor virtual
50 delete temp;
51 }
52 cabeza = nullptr;
53 cantidad = 0;
54 }

56public:
57 /**
58 * @brief Constructor
59 */
60 ListaGestion() : cabeza(nullptr), cantidad(0) {}

62 /**
63 * @brief Destructor

69 /**
70 * @brief Agrega un sensor a la lista
71 * @param sensor Puntero al sensor (será propiedad de la lista)
72 */
73 void agregarSensor(SensorBase* sensor) {
74 NodoSensor* nuevo = new NodoSensor(sensor);
75
76 if (cabeza == nullptr) {
77 cabeza = nuevo;
78 } else {
79 NodoSensor* actual = cabeza;
80 while (actual->siguiente != nullptr) {
81 actual = actual->siguiente;
82 }
83 actual->siguiente = nuevo;
84 }
85
86 cantidad++;
87 }

89 /**
90 * @brief Busca un sensor por ID
91 * @param id Identificador a buscar
92 * @return Puntero al sensor o nullptr si no existe
93 */
94 SensorBase* buscarPorId(const char* id) {
95 NodoSensor* actual = cabeza;
96 while (actual != nullptr) {
97 if (strcmp(actual->sensor->getId(), id) == 0) {
98 return actual->sensor;
99 }
100 actual = actual->siguiente;
101 }
102 return nullptr;

112 if (cantidad == 0) {

122 actual->sensor->procesarLectura(); // Llamada polimórfica
123 actual = actual->siguiente;
124 }
125 }

137
138 NodoSensor* actual = cabeza;
139 while (actual != nullptr) {
140 actual->sensor->imprimirInfo(); // Llamada polimórfica
141 actual = actual->siguiente;
142 }
143 }
