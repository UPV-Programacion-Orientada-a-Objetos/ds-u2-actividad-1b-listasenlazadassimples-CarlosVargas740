34void procesarDatoArduino(char* buffer, ListaGestion& listaGestion) {
35 char tipo[10] = "";
36 char id[20] = "";
37 char valor[20] = "";
38
39 // Dividir el string "TIPO:ID:VALOR"
40 char* token = strtok(buffer, ":");
41 if (token) strcpy(tipo, token);
42
43 token = strtok(nullptr, ":");
44 if (token) strcpy(id, token);
45
46 token = strtok(nullptr, ":");
47 if (token) strcpy(valor, token);
48
49 cout << " Dato Arduino: " << tipo << " | " << id << " | " << valor << endl;
50
51 // Buscar sensor
52 SensorBase* sensor = listaGestion.buscarPorId(id);
53
54 // Si no existe, crearlo
55 if (sensor == nullptr) {
56 if (strcmp(tipo, "TEMP") == 0) {
57 sensor = new SensorTemperatura(id, "Arduino");
58 listaGestion.agregarSensor(sensor);
59 } else if (strcmp(tipo, "PRES") == 0) {
60 sensor = new SensorPresion(id, "Arduino");
61 listaGestion.agregarSensor(sensor);
62 } else if (strcmp(tipo, "VIBR") == 0) {
63 sensor = new SensorVibracion(id, "Arduino");
64 listaGestion.agregarSensor(sensor);
65 }
66 sensor = listaGestion.buscarPorId(id);
67 }
68
69 // Agregar lectura según tipo
70 if (sensor) {
71 SensorTemperatura* temp = dynamic_cast<SensorTemperatura*>(sensor);
72 if (temp) {
73 temp->agregarLectura(atof(valor));
74 return;
75 }
76
77 SensorPresion* pres = dynamic_cast<SensorPresion*>(sensor);
78 if (pres) {
79 pres->agregarLectura(atoi(valor));
80 return;
81 }
82
83 SensorVibracion* vibr = dynamic_cast<SensorVibracion*>(sensor);
84 if (vibr) {
85 vibr->agregarLectura(atoi(valor));
86 }
87 }
88}
